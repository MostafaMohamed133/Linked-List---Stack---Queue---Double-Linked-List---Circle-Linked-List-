};

    int LengthIs() const;

    void ResetList();
    void GetNextItem(int& item);
     void Print() const;

    NodeType* head;
    int length;
    NodeType* tail;
};

#include "UnsortedType.h"

using namespace std;

UnsortedType::UnsortedType() {
    length = 0;
    head = nullptr;
    tail = nullptr;
}
UnsortedType::~UnsortedType() {
    MakeEmpty();
}

void UnsortedType::MakeEmpty() {
    NodeType* tempPtr;

    while (head !=nullptr) {
        tempPtr = head;
        head = head->next;
        delete tempPtr;
    }

    length = 0;
}

bool UnsortedType::IsFull() const {
    try {
        NodeType* newNode = new NodeType;
        delete newNode;
        return false;
    } catch (std::bad_alloc&) {
        return true;
    }
}

int UnsortedType::LengthIs() const {
    return length;
}

void UnsortedType::RetrieveItem(int& item, bool& found) {
    NodeType* temp = head;
    found = false;

    while (temp != nullptr && !found) {
        if (item == temp->data) {
            found = true;
            item = temp->data;
        } else {
            temp = temp->next;
        }
    }
}

void UnsortedType::InsertItem(int item) {
    NodeType* newNode;
    newNode->data = item;
    newNode->next = head;
    head = newNode;
    length++;
}

void UnsortedType::DeleteItem(int item) {
    NodeType* temp = head;
    NodeType* tempLocation;

    if (item == head->data) {
        tempLocation = head;
        head = head->next;
    } else {
        while (!(item == (temp->next)-> data))
            temp = temp->next;

        tempLocation = temp->next;
        temp->next = (temp->next)->next;
    }

    delete tempLocation;
    length--;
}

void UnsortedType::Print() const {
    NodeType* temp = head;

    cout << "List contents: ";
    while (temp != nullptr) {
        cout << temp->data << " ";
        temp = temp->next;
    }
    cout <<endl;
}


void UnsortedType::ResetList() {
    tail = nullptr;
}

void UnsortedType::GetNextItem(int& item) {
    if (tail == nullptr)
        tail = head;
    else
        tail = tail->next;

    item = tail->data;
}
